class InformationTheoryMetrics:
    """信息论基础度量类"""
    
    @staticmethod
    def shannon_entropy(prob_dist: np.ndarray, base: float = 2) -> float:
        """
        计算香农熵
        
        Parameters:
        -----------
        prob_dist : np.ndarray
            概率分布（归一化）
        base : float
            对数的底，默认2（单位为比特）
            
        Returns:
        --------
        float : 香农熵值
        """
        # 过滤零概率项以避免log(0)
        prob_dist = prob_dist[prob_dist > 0]
        return -np.sum(prob_dist * np.log(prob_dist)) / np.log(base)
    
    @staticmethod
    def joint_entropy(joint_prob: np.ndarray, base: float = 2) -> float:
        """
        计算联合熵 H(X,Y)
        
        Parameters:
        -----------
        joint_prob : np.ndarray
            联合概率分布 P(X,Y)
        base : float
            对数的底
            
        Returns:
        --------
        float : 联合熵值
        """
        joint_prob = joint_prob[joint_prob > 0]
        return -np.sum(joint_prob * np.log(joint_prob)) / np.log(base)
    
    @staticmethod
    def conditional_entropy(joint_prob: np.ndarray, marginal_y: np.ndarray, 
                           base: float = 2) -> float:
        """
        计算条件熵 H(X|Y) = H(X,Y) - H(Y)
        
        Parameters:
        -----------
        joint_prob : np.ndarray
            联合概率分布 P(X,Y)
        marginal_y : np.ndarray
            边缘分布 P(Y)
        base : float
            对数的底
            
        Returns:
        --------
        float : 条件熵值
        """
        H_joint = InformationTheoryMetrics.joint_entropy(joint_prob, base)
        H_y = InformationTheoryMetrics.shannon_entropy(marginal_y, base)
        return H_joint - H_y
    
    @staticmethod
    def mutual_info(joint_prob: np.ndarray, marginal_x: np.ndarray, 
                   marginal_y: np.ndarray, base: float = 2) -> float:
        """
        计算互信息 I(X;Y) = H(X) + H(Y) - H(X,Y)
        
        Parameters:
        -----------
        joint_prob : np.ndarray
            联合概率分布 P(X,Y)
        marginal_x : np.ndarray
            边缘分布 P(X)
        marginal_y : np.ndarray
            边缘分布 P(Y)
        base : float
            对数的底
            
        Returns:
        --------
        float : 互信息值
        """
        H_x = InformationTheoryMetrics.shannon_entropy(marginal_x, base)
        H_y = InformationTheoryMetrics.shannon_entropy(marginal_y, base)
        H_joint = InformationTheoryMetrics.joint_entropy(joint_prob, base)
        return H_x + H_y - H_joint
    
    @staticmethod
    def kl_divergence(p: np.ndarray, q: np.ndarray, base: float = 2) -> float:
        """
        计算KL散度 D_KL(P||Q)
        
        Parameters:
        -----------
        p : np.ndarray
            真实分布 P
        q : np.ndarray
            近似分布 Q
        base : float
            对数的底
            
        Returns:
        --------
        float : KL散度值（非负）
        """
        # 避免除零和log(0)
        mask = (p > 1e-12) & (q > 1e-12)
        return np.sum(p[mask] * np.log(p[mask] / q[mask])) / np.log(base)
    
    @staticmethod
    def js_divergence(p: np.ndarray, q: np.ndarray, base: float = 2) -> float:
        """
        计算JS散度（对称的KL散度）
        
        JS(P||Q) = 0.5*D_KL(P||M) + 0.5*D_KL(Q||M)
        其中 M = 0.5*(P+Q)
        
        Parameters:
        -----------
        p : np.ndarray
            分布 P
        q : np.ndarray
            分布 Q
        base : float
            对数的底
            
        Returns:
        --------
        float : JS散度值（有界：[0, 1] 当使用以2为底的对数）
        """
        m = 0.5 * (p + q)
        return 0.5 * InformationTheoryMetrics.kl_divergence(p, m, base) + \
               0.5 * InformationTheoryMetrics.kl_divergence(q, m, base)

class NetworkMetrics:
    """复杂网络度量类"""
    
    def __init__(self, adj_matrix: Union[np.ndarray, csr_matrix]):
        """
        初始化网络度量类
        
        Parameters:
        -----------
        adj_matrix : np.ndarray or csr_matrix
            网络的邻接矩阵（加权或无权）
        """
        if isinstance(adj_matrix, csr_matrix):
            self.adj = adj_matrix
        else:
            self.adj = csr_matrix(adj_matrix)
        
        self.n_nodes = self.adj.shape[0]
        self._degrees = None
        self._distances = None
        
    @property
    def degrees(self) -> np.ndarray:
        """节点度序列"""
        if self._degrees is None:
            self._degrees = np.array(self.adj.sum(axis=1)).flatten()
        return self._degrees
    
    def degree_distribution(self) -> Tuple[np.ndarray, np.ndarray]:
        """
        计算度分布
        
        Returns:
        --------
        unique_degrees : np.ndarray
            不同的度值
        probabilities : np.ndarray
            对应的概率
        """
        unique, counts = np.unique(self.degrees, return_counts=True)
        probs = counts / self.n_nodes
        return unique, probs
    
    def average_degree(self) -> float:
        """平均度"""
        return np.mean(self.degrees)
    
    def degree_variance(self) -> float:
        """度方差"""
        return np.var(self.degrees)
    
    def second_moment(self) -> float:
        """度的二阶矩（用于渗流分析）"""
        return np.mean(self.degrees ** 2)
    
    def clustering_coefficient(self, method: str = 'local') -> Union[np.ndarray, float]:
        """
        计算集聚系数
        
        Parameters:
        -----------
        method : str
            'local' - 返回每个节点的局部集聚系数
            'global' - 返回全局集聚系数（所有节点平均）
            'transitivity' - 返回传递性（闭合三元组比例）
            
        Returns:
        --------
        np.ndarray or float : 集聚系数
        """
        # 转换为稠密矩阵（对于稀疏图可能需要优化）
        if self.n_nodes > 1000:
            warnings.warn("Network is large, clustering coefficient computation may be slow")
        
        adj_dense = self.adj.toarray()
        
        if method == 'local':
            coeffs = np.zeros(self.n_nodes)
            for i in range(self.n_nodes):
                neighbors = np.where(adj_dense[i] > 0)[0]
                k = len(neighbors)
                if k < 2:
                    coeffs[i] = 0
                else:
                    # 邻居之间的边数
                    e = np.sum(adj_dense[np.ix_(neighbors, neighbors)]) / 2
                    coeffs[i] = 2 * e / (k * (k - 1))
            return coeffs
        
        elif method == 'global':
            coeffs = self.clustering_coefficient('local')
            return np.mean(coeffs)
        
        elif method == 'transitivity':
            # 计算三角形数量和连通三元组数量
            triangles = 0
            triplets = 0
            for i in range(self.n_nodes):
                neighbors = np.where(adj_dense[i] > 0)[0]
                k = len(neighbors)
                if k >= 2:
                    # 邻居之间的边数
                    e = np.sum(adj_dense[np.ix_(neighbors, neighbors)]) / 2
                    triangles += e
                    triplets += k * (k - 1) / 2
            return triangles / triplets if triplets > 0 else 0
        
        else:
            raise ValueError(f"Unknown method: {method}")
    
    def shortest_paths(self) -> np.ndarray:
        """
        计算所有节点对之间的最短路径
        
        Returns:
        --------
        np.ndarray : 距离矩阵
        """
        if self._distances is None:
            self._distances = shortest_path(self.adj, directed=False, unweighted=True)
        return self._distances
    
    def average_path_length(self) -> float:
        """
        计算平均路径长度
        
        Returns:
        --------
        float : 平均路径长度（仅考虑有限距离）
        """
        dist = self.shortest_paths()
        # 只考虑有限距离（不包括无穷大和0，后者是自身距离）
        finite_dist = dist[(dist > 0) & (dist < np.inf)]
        return np.mean(finite_dist) if len(finite_dist) > 0 else np.inf
    
    def diameter(self) -> int:
        """
        计算网络直径（最长最短路径）
        
        Returns:
        --------
        int : 直径
        """
        dist = self.shortest_paths()
        finite_dist = dist[dist < np.inf]
        return int(np.max(finite_dist)) if len(finite_dist) > 0 else np.inf
    
    def efficiency(self, local: bool = False) -> Union[float, np.ndarray]:
        """
        计算全局或局部效率
        
        效率定义为1/距离，对于不连通的节点对，效率为0
        
        Parameters:
        -----------
        local : bool
            如果True，返回每个节点的局部效率；否则返回全局效率
            
        Returns:
        --------
        float or np.ndarray : 效率值
        """
        dist = self.shortest_paths()
        
        if local:
            # 局部效率：每个节点的邻居子图的效率
            local_effs = np.zeros(self.n_nodes)
            for i in range(self.n_nodes):
                neighbors = np.where((dist[i] > 0) & (dist[i] < np.inf) & (dist[i] == 1))[0]
                if len(neighbors) < 2:
                    local_effs[i] = 0
                else:
                    # 邻居子图的距离
                    sub_dist = dist[np.ix_(neighbors, neighbors)]
                    # 效率矩阵
                    eff_matrix = np.zeros_like(sub_dist, dtype=float)
                    eff_matrix[sub_dist > 0] = 1.0 / sub_dist[sub_dist > 0]
                    # 局部效率是子图效率的平均（不包括对角线）
                    np.fill_diagonal(eff_matrix, 0)
                    local_effs[i] = np.sum(eff_matrix) / (len(neighbors) * (len(neighbors) - 1))
            return local_effs
        else:
            # 全局效率
            eff_matrix = np.zeros_like(dist, dtype=float)
            eff_matrix[dist > 0] = 1.0 / dist[dist > 0]
            np.fill_diagonal(eff_matrix, 0)
            return np.sum(eff_matrix) / (self.n_nodes * (self.n_nodes - 1))
    
    def centrality(self, type: str = 'degree') -> np.ndarray:
        """
        计算各种中心性度量
        
        Parameters:
        -----------
        type : str
            'degree' - 度中心性
            'betweenness' - 介数中心性（近似）
            'closeness' - 接近中心性
            'eigenvector' - 特征向量中心性（幂迭代）
            
        Returns:
        --------
        np.ndarray : 每个节点的中心性值
        """
        if type == 'degree':
            return self.degrees / (self.n_nodes - 1)
        
        elif type == 'closeness':
            dist = self.shortest_paths()
            closeness = np.zeros(self.n_nodes)
            for i in range(self.n_nodes):
                reachable = dist[i][(dist[i] > 0) & (dist[i] < np.inf)]
                if len(reachable) > 0:
                    closeness[i] = len(reachable) / np.sum(reachable)
            return closeness
        
        elif type == 'betweenness':
            # 基于Brandes算法的近似实现
            # 注意：完整实现较复杂，这里提供简化版本
            betweenness = np.zeros(self.n_nodes)
            
            for s in range(self.n_nodes):
                # BFS找到最短路径
                dist = np.full(self.n_nodes, np.inf)
                dist[s] = 0
                sigma = np.zeros(self.n_nodes)
                sigma[s] = 1
                queue = [s]
                
                # 记录前驱节点
                pred = [[] for _ in range(self.n_nodes)]
                
                # BFS
                while queue:
                    v = queue.pop(0)
                    # 找到v的邻居
                    neighbors = np.where(self.adj[v].toarray()[0] > 0)[0]
                    for w in neighbors:
                        if dist[w] == np.inf:
                            dist[w] = dist[v] + 1
                            queue.append(w)
                        if dist[w] == dist[v] + 1:
                            sigma[w] += sigma[v]
                            pred[w].append(v)
                
                # 回溯计算依赖
                delta = np.zeros(self.n_nodes)
                # 按距离从远到近处理
                nodes_by_dist = np.argsort(-dist)
                for w in nodes_by_dist:
                    if dist[w] < np.inf:
                        for v in pred[w]:
                            delta[v] += (sigma[v] / sigma[w]) * (1 + delta[w])
                        if w != s:
                            betweenness[w] += delta[w]
            
            # 归一化（无向图）
            betweenness = betweenness / ((self.n_nodes - 1) * (self.n_nodes - 2) / 2)
            return betweenness
        
        elif type == 'eigenvector':
            # 幂迭代法计算主特征向量
            # 使用归一化的邻接矩阵（按行或列）
            
            # 创建转移矩阵（列归一化）
            adj_dense = self.adj.toarray()
            col_sums = np.sum(adj_dense, axis=0)
            
            # 避免除零
            col_sums[col_sums == 0] = 1
            
            # 转移矩阵
            M = adj_dense / col_sums
            
            # 幂迭代
            v = np.random.rand(self.n_nodes)
            v = v / np.linalg.norm(v, 1)
            
            for _ in range(100):  # 最大迭代次数
                v_new = M @ v
                v_new = v_new / np.linalg.norm(v_new, 1)
                
                if np.linalg.norm(v_new - v, 1) < 1e-8:
                    break
                v = v_new
            
            return v
        
        else:
            raise ValueError(f"Unknown centrality type: {type}")

class MultiscaleEntropy:
    """多尺度熵分析类"""
    
    def __init__(self, m: int = 2, r: float = 0.15):
        """
        初始化多尺度熵分析器
        
        Parameters:
        -----------
        m : int
            模式长度
        r : float
            容差（相对于标准差的比例）
        """
        self.m = m
        self.r = r
    
    def _count_matches(self, templates: np.ndarray, r: float) -> float:
        """
        计算模板匹配的数量
        
        Parameters:
        -----------
        templates : np.ndarray
            模板矩阵，每行是一个m维模板
        r : float
            容差
            
        Returns:
        --------
        float : 平均匹配数
        """
        N = len(templates)
        if N < 2:
            return 0.0
        
        # 计算所有模板对之间的距离
        # 使用广播和向量化以提高效率
        # 对于大数据集，可能需要分块计算
        
        matches = 0
        for i in range(N):
            # 计算第i个模板与其他所有模板的切比雪夫距离
            dists = np.max(np.abs(templates - templates[i]), axis=1)
            # 统计距离小于等于r的数量（包括自身）
            matches += np.sum(dists <= r)
        
        # 返回平均匹配数
        return matches / N
    
    def sample_entropy(self, signal: np.ndarray) -> float:
        """
        计算样本熵（Sample Entropy）
        
        Parameters:
        -----------
        signal : np.ndarray
            输入时间序列
            
        Returns:
        --------
        float : 样本熵值
        """
        N = len(signal)
        if N < self.m + 1:
            return np.nan
        
        # 计算标准差并设置容差
        sd = np.std(signal)
        r = self.r * sd
        
        if r < 1e-10:
            return 0.0
        
        # 构建m维模板
        def build_templates(data, m):
            N = len(data)
            templates = np.zeros((N - m, m))
            for i in range(N - m):
                templates[i] = data[i:i+m]
            return templates
        
        # 计算m维模板的匹配数
        templates_m = build_templates(signal, self.m)
        A = self._count_matches(templates_m, r)
        
        # 计算m+1维模板的匹配数
        templates_m1 = build_templates(signal, self.m + 1)
        B = self._count_matches(templates_m1, r)
        
        # 样本熵 = -ln(B/A)
        if A < 1e-10 or B < 1e-10:
            return np.nan
        
        return -np.log(B / A)
    
    def approximate_entropy(self, signal: np.ndarray) -> float:
        """
        计算近似熵（Approximate Entropy）
        
        Parameters:
        -----------
        signal : np.ndarray
            输入时间序列
            
        Returns:
        --------
        float : 近似熵值
        """
        N = len(signal)
        if N < self.m + 1:
            return np.nan
        
        sd = np.std(signal)
        r = self.r * sd
        
        if r < 1e-10:
            return 0.0
        
        def phi_m(data, m):
            N = len(data)
            count = np.zeros(N - m + 1)
            
            for i in range(N - m + 1):
                template_i = data[i:i+m]
                matches = 0
                for j in range(N - m + 1):
                    template_j = data[j:j+m]
                    if np.max(np.abs(template_i - template_j)) <= r:
                        matches += 1
                count[i] = matches / (N - m + 1)
            
            return np.mean(np.log(count[count > 0]))
        
        phi_m_val = phi_m(signal, self.m)
        phi_m1_val = phi_m(signal, self.m + 1)
        
        return phi_m_val - phi_m1_val
    
    def multiscale_entropy(self, signal: np.ndarray, max_scale: int = 20) -> np.ndarray:
        """
        计算多尺度熵（Multiscale Entropy）
        
        Parameters:
        -----------
        signal : np.ndarray
            输入时间序列
        max_scale : int
            最大尺度
            
        Returns:
        --------
        np.ndarray : 每个尺度的样本熵值
        """
        mse_values = np.zeros(max_scale)
        
        for tau in range(1, max_scale + 1):
            # 粗粒化
            coarse_signal = self._coarse_grain(signal, tau)
            
            # 计算样本熵
            if len(coarse_signal) >= self.m + 1:
                # 临时调整容差（因为粗粒化后标准差可能不同）
                sd = np.std(coarse_signal)
                r = self.r * sd
                
                # 使用原始的r比例计算
                mse_values[tau-1] = self._sample_entropy_fast(coarse_signal, self.m, r)
            else:
                mse_values[tau-1] = np.nan
        
        return mse_values
    
    def _coarse_grain(self, signal: np.ndarray, scale: int) -> np.ndarray:
        """
        对时间序列进行粗粒化
        
        Parameters:
        -----------
        signal : np.ndarray
            原始时间序列
        scale : int
            尺度因子
            
        Returns:
        --------
        np.ndarray : 粗粒化后的时间序列
        """
        N = len(signal)
        n_windows = N // scale
        coarse = np.zeros(n_windows)
        
        for i in range(n_windows):
            coarse[i] = np.mean(signal[i*scale:(i+1)*scale])
        
        return coarse
    
    def _sample_entropy_fast(self, signal: np.ndarray, m: int, r: float) -> float:
        """
        快速样本熵计算（简化版）
        
        使用KD树或分块策略可以进一步加速
        """
        N = len(signal)
        if N < m + 1:
            return np.nan
        
        # 构建模板
        def build_templates(data, m):
            N = len(data)
            return np.array([data[i:i+m] for i in range(N - m)])
        
        templates_m = build_templates(signal, m)
        templates_m1 = build_templates(signal, m + 1)
        
        # 计算匹配数
        def count_matches(templates, r):
            N = len(templates)
            if N < 2:
                return 0, 0
            
            total_matches = 0
            for i in range(N):
                dists = np.max(np.abs(templates - templates[i]), axis=1)
                matches = np.sum(dists <= r)
                total_matches += matches
            
            return total_matches, N
        
        matches_m, n_m = count_matches(templates_m, r)
        matches_m1, n_m1 = count_matches(templates_m1, r)
        
        # 计算熵
        if n_m == 0 or n_m1 == 0:
            return np.nan
        
        A = matches_m / n_m
        B = matches_m1 / n_m1
        
        if A < 1e-10 or B < 1e-10:
            return np.nan
        
        return -np.log(B / A)

class StatisticalPhysics:
    """统计物理计算类"""
    
    @staticmethod
    def partition_function(energies: np.ndarray, temperature: float, 
                          boltzmann_constant: float = 1.0) -> float:
        """
        计算正则系综配分函数
        
        Z = sum_i exp(-beta * E_i)
        
        Parameters:
        -----------
        energies : np.ndarray
            能级
        temperature : float
            温度
        boltzmann_constant : float
            玻尔兹曼常数
            
        Returns:
        --------
        float : 配分函数值
        """
        beta = 1.0 / (boltzmann_constant * temperature)
        return np.sum(np.exp(-beta * energies))
    
    @staticmethod
    def free_energy(energies: np.ndarray, temperature: float,
                   boltzmann_constant: float = 1.0) -> float:
        """
        计算亥姆霍兹自由能
        
        F = -kT * ln(Z)
        
        Parameters:
        -----------
        energies : np.ndarray
            能级
        temperature : float
            温度
        boltzmann_constant : float
            玻尔兹曼常数
            
        Returns:
        --------
        float : 自由能
        """
        Z = StatisticalPhysics.partition_function(energies, temperature, boltzmann_constant)
        return -boltzmann_constant * temperature * np.log(Z)
    
    @staticmethod
    def internal_energy(energies: np.ndarray, temperature: float,
                       boltzmann_constant: float = 1.0) -> float:
        """
        计算内能
        
        U = <E> = sum_i E_i * P_i = -d(ln Z)/d(beta)
        
        Parameters:
        -----------
        energies : np.ndarray
            能级
        temperature : float
            温度
        boltzmann_constant : float
            玻尔兹曼常数
            
        Returns:
        --------
        float : 内能
        """
        beta = 1.0 / (boltzmann_constant * temperature)
        weights = np.exp(-beta * energies)
        Z = np.sum(weights)
        return np.sum(energies * weights) / Z
    
    @staticmethod
    def entropy(energies: np.ndarray, temperature: float,
               boltzmann_constant: float = 1.0) -> float:
        """
        计算热力学熵
        
        S = (U - F) / T = -k * sum_i P_i * ln(P_i)
        
        Parameters:
        -----------
        energies : np.ndarray
            能级
        temperature : float
            温度
        boltzmann_constant : float
            玻尔兹曼常数
            
        Returns:
        --------
        float : 熵
        """
        beta = 1.0 / (boltzmann_constant * temperature)
        weights = np.exp(-beta * energies)
        Z = np.sum(weights)
        probs = weights / Z
        
        # S = -k * sum(p * ln(p))
        S = -boltzmann_constant * np.sum(probs * np.log(probs))
        return S
    
    @staticmethod
    def heat_capacity(energies: np.ndarray, temperature: float,
                     boltzmann_constant: float = 1.0) -> float:
        """
        计算热容
        
        C = dU/dT = (<E^2> - <E>^2) / (kT^2)
        
        Parameters:
        -----------
        energies : np.ndarray
            能级
        temperature : float
            温度
        boltzmann_constant : float
            玻尔兹曼常数
            
        Returns:
        --------
        float : 热容
        """
        beta = 1.0 / (boltzmann_constant * temperature)
        weights = np.exp(-beta * energies)
        Z = np.sum(weights)
        
        avg_E = np.sum(energies * weights) / Z
        avg_E2 = np.sum(energies**2 * weights) / Z
        
        variance = avg_E2 - avg_E**2
        return variance / (boltzmann_constant * temperature**2)
    
    @staticmethod
    def ising_2d_exact(temperature: float, J: float = 1.0, 
                       boltzmann_constant: float = 1.0) -> Dict[str, float]:
        """
        计算二维Ising模型的Onsager精确解
        
        Parameters:
        -----------
        temperature : float
            温度
        J : float
            耦合常数
        boltzmann_constant : float
            玻尔兹曼常数
            
        Returns:
        --------
        Dict[str, float] : 包含各种热力学量的字典
        """
        beta = 1.0 / (boltzmann_constant * temperature)
        
        # 临界点
        T_c = 2 * J / (boltzmann_constant * np.log(1 + np.sqrt(2)))
        
        # 定义被积函数
        def integrand(theta, beta, J):
            k = 1 / np.sinh(2 * beta * J)**2
            return np.log(1 + np.sqrt(1 - k * np.sin(theta)**2))
        
        # 数值积分计算自由能
        from scipy.integrate import quad
        k = 1 / np.sinh(2 * beta * J)**2
        
        if k <= 1:  # T >= T_c
            integral, _ = quad(integrand, 0, np.pi/2, args=(beta, J))
            free_energy = -J * np.cosh(2 * beta * J) / np.tanh(2 * beta * J) - boltzmann_constant * temperature * np.log(2) / np.pi * integral
        else:
            # 低温展开
            free_energy = -J - boltzmann_constant * temperature * np.log(2)
        
        # 数值微分计算其他量
        dT = temperature * 0.001
        T_plus = temperature + dT
        T_minus = temperature - dT
        
        # 简化的热容估计
        if abs(temperature - T_c) < 0.1 * T_c:
            heat_capacity = np.log(np.abs(temperature - T_c) / T_c) if temperature != T_c else 10.0
        else:
            heat_capacity = 0.5 * boltzmann_constant * (2 * beta * J / np.sinh(2 * beta * J))**2
        
        return {
            'temperature': temperature,
            'critical_temperature': T_c,
            'free_energy_per_site': free_energy,
            'heat_capacity_per_site': heat_capacity,
            'beta': beta,
            'sinh_2betaJ': np.sinh(2 * beta * J)
        }

def generate_power_law(n: int, gamma: float, k_min: int = 1, k_max: Optional[int] = None) -> np.ndarray:
    """
    生成幂律分布的度序列
    
    Parameters:
    -----------
    n : int
        节点数
    gamma : float
        幂指数
    k_min : int
        最小度
    k_max : int, optional
        最大度，默认为n-1
        
    Returns:
    --------
    np.ndarray : 度序列
    """
    if k_max is None:
        k_max = n - 1
    
    # 离散幂律采样
    degrees = np.arange(k_min, k_max + 1)
    probs = degrees ** (-gamma)
    probs = probs / np.sum(probs)
    
    # 采样
    degree_seq = np.random.choice(degrees, size=n, p=probs)
    
    # 确保度序列可图化（度和为偶数）
    if np.sum(degree_seq) % 2 != 0:
        # 随机调整一个节点的度
        idx = np.random.randint(n)
        if degree_seq[idx] > k_min:
            degree_seq[idx] -= 1
        else:
            degree_seq[idx] += 1
    
    return degree_seq


def configuration_model(degree_seq: np.ndarray) -> csr_matrix:
    """
    使用配置模型生成随机网络
    
    Parameters:
    -----------
    degree_seq : np.ndarray
        度序列
        
    Returns:
    --------
    csr_matrix : 邻接矩阵（稀疏）
    """
    n = len(degree_seq)
    
    # 创建stubs列表
    stubs = []
    for i, k in enumerate(degree_seq):
        stubs.extend([i] * int(k))
    
    # 随机配对
    np.random.shuffle(stubs)
    
    # 构建边列表
    edges = []
    for i in range(0, len(stubs) - 1, 2):
        u, v = stubs[i], stubs[i+1]
        if u != v:  # 排除自环
            edges.append((u, v))
            edges.append((v, u))  # 无向图
    
    # 构建稀疏邻接矩阵
    if len(edges) > 0:
        rows, cols = zip(*edges)
        data = np.ones(len(rows))
        adj = csr_matrix((data, (rows, cols)), shape=(n, n))
    else:
        adj = csr_matrix((n, n))
    
    return adj


def erdos_renyi_graph(n: int, p: float) -> csr_matrix:
    """
    生成Erdős-Rényi随机图 G(n,p)
    
    Parameters:
    -----------
    n : int
        节点数
    p : float
        连边概率
        
    Returns:
    --------
    csr_matrix : 邻接矩阵
    """
    # 使用二项分布采样边数
    n_possible = n * (n - 1) // 2
    n_edges = np.random.binomial(n_possible, p)
    
    # 随机选择边
    edges = set()
    while len(edges) < n_edges:
        u = np.random.randint(0, n)
        v = np.random.randint(0, n)
        if u < v:
            edges.add((u, v))
    
    # 构建邻接矩阵
    if len(edges) > 0:
        rows, cols = zip(*edges)
        # 对称矩阵
        row_idx = list(rows) + list(cols)
        col_idx = list(cols) + list(rows)
        data = np.ones(len(row_idx))
        adj = csr_matrix((data, (row_idx, col_idx)), shape=(n, n))
    else:
        adj = csr_matrix((n, n))
    
    return adj

def demo_multiscale_entropy():
    """演示多尺度熵分析"""
    print("\n" + "="*60)
    print("多尺度熵分析演示")
    print("="*60)
    
    np.random.seed(42)
    
    # 生成不同类型的时间序列
    n = 2000
    
    print("\n1. 生成测试信号")
    
    # 白噪声（高斯噪声）
    white_noise = np.random.randn(n)
    print(f"白噪声: 标准差={np.std(white_noise):.3f}")
    
    # 1/f噪声（粉红噪声）- 使用近似方法
    # 通过在频域设置适当的功率谱然后逆傅里叶变换
    freqs = np.fft.fftfreq(n)
    freqs[0] = 1e-10  # 避免除零
    pink_spectrum = 1 / np.sqrt(np.abs(freqs))
    pink_spectrum[0] = 0  # DC分量为0
    
    # 随机相位
    phases = np.random.uniform(0, 2*np.pi, n)
    complex_spectrum = pink_spectrum * np.exp(1j * phases)
    pink_noise = np.real(np.fft.ifft(complex_spectrum))
    pink_noise = pink_noise / np.std(pink_noise)  # 归一化
    print(f"1/f噪声: 标准差={np.std(pink_noise):.3f}")
    
    # 确定性信号（正弦波加噪声）
    t = np.linspace(0, 10*2*np.pi, n)
    periodic = np.sin(t) + 0.3 * np.random.randn(n)
    print(f"周期信号: 标准差={np.std(periodic):.3f}")
    
    # Lorenz混沌系统（简化近似）
    # 这里用一个简单的非线性映射近似
    lorenz_approx = np.zeros(n)
    lorenz_approx[0] = 0.1
    for i in range(1, n):
        lorenz_approx[i] = 3.8 * lorenz_approx[i-1] * (1 - lorenz_approx[i-1])
    lorenz_approx = (lorenz_approx - np.mean(lorenz_approx)) / np.std(lorenz_approx)
    print(f"混沌信号: 标准差={np.std(lorenz_approx):.3f}")
    
    # 多尺度熵分析
    print("\n2. 多尺度熵分析")
    
    mse = MultiscaleEntropy(m=2, r=0.15)
    max_scale = 20
    
    signals = {
        '白噪声': white_noise,
        '1/f噪声': pink_noise,
        '周期信号': periodic,
        '混沌信号': lorenz_approx
    }
    
    results = {}
    
    for name, signal in signals.items():
        print(f"\n  分析 {name}...")
        mse_values = mse.multiscale_entropy(signal, max_scale)
        results[name] = mse_values
        
        # 统计信息
        valid_mse = mse_values[~np.isnan(mse_values)]
        if len(valid_mse) > 0:
            print(f"    MSE范围: [{np.min(valid_mse):.3f}, {np.max(valid_mse):.3f}]")
            print(f"    MSE均值: {np.mean(valid_mse):.3f}")
    
    # 分析结果解释
    print("\n3. 结果分析")
    print("-" * 50)
    
    print("""
    理论预期与观察：
    
    1. 白噪声：
       - 所有尺度上样本熵应该较高且相对平坦
       - 因为白噪声在所有频率上都有能量，没有相关性
    
    2. 1/f噪声（粉红噪声）：
       - 应该在多个尺度上保持较高的复杂性
       - 由于长程相关性，熵值随尺度下降较慢
    
    3. 周期信号：
       - 在小尺度上可能有较低的熵（明显的模式）
       - 在大尺度上可能表现出周期性导致的特定模式
    
    4. 混沌信号：
       - 应该在中等尺度上表现出较高的复杂性
       - 由于确定性混沌的短期可预测性和长期不可预测性
    """)
    
    print("\n实际计算结果特征：")
    for name, mse_values in results.items():
        valid = mse_values[~np.isnan(mse_values)]
        if len(valid) > 5:
            trend = np.polyfit(range(len(valid)), valid, 1)[0]
            print(f"  {name}: 平均MSE={np.mean(valid):.3f}, "
                  f"趋势={'下降' if trend < -0.01 else '上升' if trend > 0.01 else '平稳'}")
    
    return results

def plot_network_comparison():
    """绘制不同网络的比较"""
    fig, axes = plt.subplots(2, 3, figsize=(15, 10))
    
    n = 100
    
    # 1. 规则网络（环形）
    ax = axes[0, 0]
    ax.set_xlim(-1.5, 1.5)
    ax.set_ylim(-1.5, 1.5)
    ax.set_aspect('equal')
    ax.axis('off')
    
    theta = np.linspace(0, 2*np.pi, n, endpoint=False)
    x = np.cos(theta)
    y = np.sin(theta)
    
    # 每个节点连接k个最近邻
    k = 4
    for i in range(n):
        ax.plot(x[i], y[i], 'o', markersize=4, color='steelblue')
        for j in range(1, k//2 + 1):
            idx1 = (i + j) % n
            idx2 = (i - j) % n
            ax.plot([x[i], x[idx1]], [y[i], y[idx1]], '-', color='gray', alpha=0.5, linewidth=0.5)
            ax.plot([x[i], x[idx2]], [y[i], y[idx2]], '-', color='gray', alpha=0.5, linewidth=0.5)
    
    ax.set_title('规则网络 (环形)\n高集聚, 长路径', fontsize=11)
    
    # 2. 随机网络
    ax = axes[0, 1]
    ax.set_xlim(-1.5, 1.5)
    ax.set_ylim(-1.5, 1.5)
    ax.set_aspect('equal')
    ax.axis('off')
    
    p = 0.05
    ax.set_title(f'随机网络 (Erdős-Rényi)\np={p}, 低集聚, 短路径', fontsize=11)
    
    # 随机位置
    np.random.seed(42)
    x_pos = np.random.randn(n)
    y_pos = np.random.randn(n)
    # 归一化
    x_pos = (x_pos - np.min(x_pos)) / (np.max(x_pos) - np.min(x_pos)) * 2 - 1
    y_pos = (y_pos - np.min(y_pos)) / (np.max(y_pos) - np.min(y_pos)) * 2 - 1
    
    # 生成边
    er_graph = erdos_renyi_graph(n, p)
    er_dense = er_graph.toarray()
    
    for i in range(n):
        for j in range(i+1, n):
            if er_dense[i, j] > 0:
                ax.plot([x_pos[i], x_pos[j]], [y_pos[i], y_pos[j]], 
                       '-', color='gray', alpha=0.3, linewidth=0.3)
    
    for i in range(n):
        ax.plot(x_pos[i], y_pos[i], 'o', markersize=3, color='coral', alpha=0.7)
    
    # 3. 小世界网络
    ax = axes[0, 2]
    ax.set_xlim(-1.5, 1.5)
    ax.set_ylim(-1.5, 1.5)
    ax.set_aspect('equal')
    ax.axis('off')
    
    # Watts-Strogatz模型：环形+重连
    k = 4
    p_rewire = 0.2
    ax.set_title(f'小世界网络 (Watts-Strogatz)\np={p_rewire}, 高集聚, 短路径', fontsize=11)
    
    theta = np.linspace(0, 2*np.pi, n, endpoint=False)
    x_ws = np.cos(theta)
    y_ws = np.sin(theta)
    
    # 创建小世界网络
    # 从规则网络开始
    adj_ws = np.zeros((n, n))
    for i in range(n):
        for j in range(1, k//2 + 1):
            idx = (i + j) % n
            adj_ws[i, idx] = 1
            adj_ws[idx, i] = 1
    
    # 随机重连
    np.random.seed(123)
    for i in range(n):
        for j in range(i+1, n):
            if adj_ws[i, j] == 1:
                if np.random.rand() < p_rewire:
                    # 移除原边
                    adj_ws[i, j] = 0
                    adj_ws[j, i] = 0
                    # 添加新边（避免重复和自环）
                    attempts = 0
                    while attempts < 100:
                        new_j = np.random.randint(n)
                        if new_j != i and adj_ws[i, new_j] == 0:
                            adj_ws[i, new_j] = 1
                            adj_ws[new_j, i] = 1
                            break
                        attempts += 1
    
    # 绘制网络
    for i in range(n):
        for j in range(i+1, n):
            if adj_ws[i, j] > 0:
                ax.plot([x_ws[i], x_ws[j]], [y_ws[i], y_ws[j]], 
                       '-', color='gray', alpha=0.2, linewidth=0.3)
    
    for i in range(n):
        ax.plot(x_ws[i], y_ws[i], 'o', markersize=3, color='mediumseagreen', alpha=0.7)
    
    # 下半部分：度分布和指标对比
    
    # 4. 度分布对比
    ax = axes[1, 0]
    
    # ER网络度分布
    nm_er = NetworkMetrics(erdos_renyi_graph(1000, 0.01))
    degrees_er, probs_er = nm_er.degree_distribution()
    ax.plot(degrees_er, probs_er, 'o-', label='Erdős-Rényi', color='steelblue', markersize=4)
    
    # 无标度网络度分布
    degree_seq = generate_power_law(1000, 2.5, k_min=2)
    nm_sf = NetworkMetrics(configuration_model(degree_seq))
    degrees_sf, probs_sf = nm_sf.degree_distribution()
    
    # 对数坐标
    ax_log = ax
    ax_log.loglog(degrees_sf[probs_sf > 0], probs_sf[probs_sf > 0], 's-', 
                  label='无标度 (幂律)', color='coral', markersize=4)
    
    ax_log.set_xlabel('度 k', fontsize=10)
    ax_log.set_ylabel('P(k)', fontsize=10)
    ax_log.set_title('度分布对比', fontsize=11)
    ax_log.legend(fontsize=9)
    ax_log.grid(True, alpha=0.3, which='both')
    
    # 5. 网络指标对比
    ax = axes[1, 1]
    
    # 对比不同网络类型
    n = 200
    
    networks = {
        '规则网络': None,  # 特殊处理
        '随机网络': erdos_renyi_graph(n, 0.03),
        '小世界网络': None,  # 需要特殊生成
        '无标度网络': configuration_model(generate_power_law(n, 2.5, k_min=2))
    }
    
    # 简化的指标对比（模拟数据，实际应计算）
    metrics = ['集聚系数', '平均路径长度', '效率']
    
    # 使用示例数据
    data = {
        '规则网络': [0.5, 15.0, 0.15],
        '随机网络': [0.03, 3.5, 0.35],
        '小世界网络': [0.4, 4.0, 0.32],
        '无标度网络': [0.15, 3.0, 0.38]
    }
    
    x = np.arange(len(metrics))
    width = 0.2
    
    colors = ['steelblue', 'coral', 'mediumseagreen', 'gold']
    
    for i, (network, values) in enumerate(data.items()):
        offset = (i - 1.5) * width
        bars = ax.bar(x + offset, values, width, label=network, color=colors[i], alpha=0.8)
    
    ax.set_xlabel('度量指标', fontsize=10)
    ax.set_ylabel('值', fontsize=10)
    ax.set_title('不同网络类型的拓扑指标对比', fontsize=11)
    ax.set_xticks(x)
    ax.set_xticklabels(metrics, fontsize=9)
    ax.legend(fontsize=8)
    ax.grid(True, alpha=0.3, axis='y')
    
    # 6. 多尺度熵曲线示例
    ax = axes[1, 2]
    
    # 模拟的多尺度熵曲线
    scales = np.arange(1, 21)
    
    # 白噪声：高且平坦
    mse_white = 1.8 - 0.02 * scales + np.random.normal(0, 0.05, 20)
    
    # 1/f噪声：中等，缓慢下降
    mse_pink = 1.5 - 0.03 * scales + np.random.normal(0, 0.03, 20)
    
    # 周期信号：低且可能有振荡
    mse_periodic = 0.8 + 0.3 * np.sin(scales / 3) + np.random.normal(0, 0.04, 20)
    
    # 混沌信号：中等复杂
    mse_chaos = 1.2 - 0.01 * scales + np.random.normal(0, 0.06, 20)
    
    ax.plot(scales, mse_white, 'o-', label='白噪声', color='steelblue', markersize=4)
    ax.plot(scales, mse_pink, 's-', label='1/f噪声', color='coral', markersize=4)
    ax.plot(scales, mse_periodic, '^-', label='周期信号', color='mediumseagreen', markersize=4)
    ax.plot(scales, mse_chaos, 'd-', label='混沌信号', color='gold', markersize=4)
    
    ax.set_xlabel('时间尺度 τ', fontsize=10)
    ax.set_ylabel('样本熵 SampEn', fontsize=10)
    ax.set_title('多尺度熵分析示例', fontsize=11)
    ax.legend(fontsize=9)
    ax.grid(True, alpha=0.3)
    ax.set_xlim(0, 21)
    
    plt.tight_layout()
    plt.savefig('complexity_measures_overview.png', dpi=150, bbox_inches='tight')
    plt.show()
    print("综合图表已保存为 'complexity_measures_overview.png'")
